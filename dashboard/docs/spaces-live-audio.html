Quick Spaces Listing

(() => {
  const spaces = [];
  document.querySelectorAll('[data-testid="SpaceCard"], a[href*="/spaces/"]').forEach(card => {
    spaces.push({
      title: card.textContent?.trim()?.substring(0, 80) || '',
      link: card.href || card.querySelector('a')?.href || '',
    });
  });
  console.table(spaces);
})();
